          led = 2; // hier definieer je een op welke plaats een led is aangesloten
          
          void setup () {
            pinMode(led, OUTPUT); // hier zeg je of de led een input of een output is
          }
    
          void loop () {
            // hieronder een korte code die een led laat knipperen
            digitalWrite(led, HIGH); // led aan
            delay(10); // een wacht periode
            digitalWrite(led, LOW); // led uit
            delay(10); // wacht periode
          }
